132 template <DjiMotorType motor_type>
133 void DjiMotor<motor_type>::SetCurrent(i16 current) {
134  // 限幅到电机能接受的最大电流
135  current = modules::algorithm::utils::absConstrain(current, DjiMotorProperties<motor_type>::kCurrentBound);
136  // 根据这个电机所属的can总线(this->can_)和电机ID(this->id_)找到对应的发送缓冲区，写入电流值
137  DjiMotorProperties<motor_type>::tx_buf_[this->can_].at((this->id_ - 1) * 2) = (current >> 8) & 0xff;
138  DjiMotorProperties<motor_type>::tx_buf_[this->can_].at((this->id_ - 1) * 2 + 1) = current & 0xff;
139  // 根据电机ID判断缓冲区前八位需要发送，还是后八位需要发送，把对应的标志位置1
140  if (1 <= this->id_ && this->id_ <= 4) {
141  DjiMotorProperties<motor_type>::tx_buf_[this->can_].at(16) = 1;
142  } else if (5 <= this->id_ && this->id_ <= 8) {
143  DjiMotorProperties<motor_type>::tx_buf_[this->can_].at(17) = 1;
144  }
145 }

150 template <DjiMotorType motor_type>
151 void DjiMotor<motor_type>::SendCommand() {
152  for (auto &buf : DjiMotorProperties<motor_type>::tx_buf_) {
153  if (buf.second.at(16) == 1) {
154  buf.first->Write(DjiMotorProperties<motor_type>::kControlId[0], buf.second.data(), 8);
155  buf.second.at(16) = 0;
156  }
157  if (buf.second.at(17) == 1) {
158  buf.first->Write(DjiMotorProperties<motor_type>::kControlId[1], buf.second.data() + 8, 8);
159  buf.second.at(17) = 0;
160  }
161  }
162 }
