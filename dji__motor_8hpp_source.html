138 template <DjiMotorType motor_type>
139 DjiMotor<motor_type>::DjiMotor(bsp::CanBase &can, u16 id)
140  : DjiMotorBase(can, DjiMotorProperties<motor_type>::kRxIdBase + id) {
141  // 检查有没有构造过ID一样的电机
142  for (const auto &motor : DjiMotorBase::motor_list_) {
143  if (motor->id_ == id) {
144  modules::exception::ThrowException(modules::exception::Exception::kValueError); // 电机ID冲突
145  }
146  }
147  // 构造的电机对象对应CAN总线的发送缓冲区还未创建，就创建一个
148  if (DjiMotorProperties<motor_type>::tx_buf_.find(&can) == DjiMotorProperties<motor_type>::tx_buf_.end()) {
149  DjiMotorProperties<motor_type>::tx_buf_.insert({&can, {0}});
150  }
151  // 把自己加到电机对象链表里
152  this->id_ = id;
153  this->tx_buf_ = &DjiMotorProperties<motor_type>::tx_buf_[&can];
154  memcpy(this->control_id_, DjiMotorProperties<motor_type>::kControlId, sizeof(this->control_id_));
155  DjiMotorBase::motor_list_.push_back(this);
156 }

164 template <DjiMotorType motor_type>
165 void DjiMotor<motor_type>::SetCurrent(i16 current) {
166  // 限幅到电机能接受的最大电流
167  current = irobot_ec::modules::algorithm::utils::absConstrain(current, DjiMotorProperties<motor_type>::kCurrentBound);
168  // 根据这个电机所属的can总线(this->can_)和电机ID(this->id_)找到对应的发送缓冲区
169  this->tx_buf_->at((this->id_ - 1) * 2) = (current >> 8) & 0xff;
170  this->tx_buf_->at((this->id_ - 1) * 2 + 1) = current & 0xff;
171  // 根据电机ID判断缓冲区前八位需要发送，还是后八位需要发送，把对应的标志位置1
172  if (1 <= this->id_ && this->id_ <= 4) {
173  this->tx_buf_->at(16) = 1;
174  } else if (5 <= this->id_ && this->id_ <= 8) {
175  this->tx_buf_->at(17) = 1;
176  }
177 }

186  this->encoder_ = (msg->data[0] << 8) | msg->data[1];
187  this->rpm_ = (msg->data[2] << 8) | msg->data[3];
188  this->current_ = (msg->data[4] << 8) | msg->data[5];
