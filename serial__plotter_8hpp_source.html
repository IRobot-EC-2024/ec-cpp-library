60 void irobot_ec::modules::SerialPlotter::Update() {
61  if (this->variable_list_.empty()) {
62  return;
63  }
64 
65  this->buffer_.clear();
66 
67  for (auto it = this->variable_list_.begin(); it != this->variable_list_.end(); it++) {
68  std::stringstream ss;
69 
70  if (it->first == typeid(int8_t)) {
71  this->buffer_ += std::to_string(*((int8_t *)it->second));
72  } else if (it->first == typeid(uint8_t)) {
73  this->buffer_ += std::to_string(*((uint8_t *)it->second));
74  } else if (it->first == typeid(int16_t)) {
75  this->buffer_ += std::to_string(*((int16_t *)it->second));
76  } else if (it->first == typeid(uint16_t)) {
77  this->buffer_ += std::to_string(*((uint16_t *)it->second));
78  } else if (it->first == typeid(int32_t)) {
79  this->buffer_ += std::to_string(*((int32_t *)it->second));
80  } else if (it->first == typeid(uint32_t)) {
81  this->buffer_ += std::to_string(*((uint32_t *)it->second));
82  } else if (it->first == typeid(int64_t)) {
83  this->buffer_ += std::to_string(*((int64_t *)it->second));
84  } else if (it->first == typeid(uint64_t)) {
85  this->buffer_ += std::to_string(*((uint64_t *)it->second));
86  } else if (it->first == typeid(fp32)) {
87  ss << std::fixed << std::setprecision(7) << *((fp32 *)it->second);
88  this->buffer_ += ss.str();
89  } else if (it->first == typeid(fp64)) {
90  ss << std::fixed << std::setprecision(7) << *((fp64 *)it->second);
